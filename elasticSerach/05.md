-------------------------------------------
# Étape 1 - Installation de Kibana
-------------------------------------------

```bash
# Mise à jour du système
sudo apt update

# Installation d'Elasticsearch
sudo apt install elasticsearch

# Activation et démarrage d'Elasticsearch
sudo systemctl enable elasticsearch
sudo systemctl start elasticsearch
sudo systemctl status elasticsearch

# Mise à jour d'Elasticsearch (si nécessaire)
sudo apt upgrade elasticsearch

# Ajout de la clé GPG et du dépôt Elasticsearch
wget -qO - https://artifacts.elastic.co/GPG-KEY-elasticsearch | sudo gpg --dearmor -o /usr/share/keyrings/elasticsearch-keyring.gpg
echo "deb [signed-by=/usr/share/keyrings/elasticsearch-keyring.gpg] https://artifacts.elastic.co/packages/8.x/apt stable main" | sudo tee /etc/apt/sources.list.d/elastic-8.x.list
sudo apt update

# Installation de Kibana
sudo apt install kibana

# Configuration de Kibana
sudo nano /etc/kibana/kibana.yml
# Modifiez les paramètres suivants dans le fichier :
# server.port: 5601
# server.host: "localhost"
# elasticsearch.hosts: ["http://localhost:9200"]

# Activation et démarrage de Kibana
sudo systemctl enable kibana
sudo systemctl start kibana
sudo systemctl status kibana

# Création du jeton d'inscription pour Kibana
sudo /usr/share/elasticsearch/bin/elasticsearch-create-enrollment-token -s kibana

# Obtention du code de vérification de Kibana
sudo /usr/share/kibana/bin/kibana-verification-code

# Réinitialisation du mot de passe pour l'utilisateur 'elastic'
sudo /usr/share/elasticsearch/bin/elasticsearch-reset-password -u elastic
```
